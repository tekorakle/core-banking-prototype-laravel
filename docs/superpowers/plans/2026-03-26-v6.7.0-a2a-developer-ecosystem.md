- [ ] **Step 3: Create the A2ATaskState enum**

```php
<?php

declare(strict_types=1);

namespace App\Domain\AgentProtocol\Enums;

/**
 * A2A Task lifecycle states per the A2A protocol specification.
 *
 * @see https://a2a-protocol.org/latest/specification/#task-object
 */
enum A2ATaskState: string
{
    case SUBMITTED = 'submitted';
    case WORKING = 'working';
    case INPUT_REQUIRED = 'input-required';
    case COMPLETED = 'completed';
    case CANCELED = 'canceled';
    case FAILED = 'failed';

    public function isTerminal(): bool
    {
        return match ($this) {
            self::COMPLETED, self::CANCELED, self::FAILED => true,
            default => false,
        };
    }

    /**
     * @return list<self>
     */
    public function allowedTransitions(): array
    {
        return match ($this) {
            self::SUBMITTED => [self::WORKING, self::CANCELED, self::FAILED],
            self::WORKING => [self::INPUT_REQUIRED, self::COMPLETED, self::CANCELED, self::FAILED],
            self::INPUT_REQUIRED => [self::WORKING, self::CANCELED, self::FAILED],
            self::COMPLETED, self::CANCELED, self::FAILED => [],
        };
    }

    public function canTransitionTo(self $target): bool
    {
        return in_array($target, $this->allowedTransitions(), true);
    }
}
```

```php
<?php

declare(strict_types=1);

namespace App\Domain\AgentProtocol\Models;

use App\Domain\AgentProtocol\Enums\A2ATaskState;
use Illuminate\Database\Eloquent\Builder;
use Illuminate\Database\Eloquent\Concerns\HasUuids;
use Illuminate\Database\Eloquent\Model;

/**
 * A2A Task — represents an async unit of work between two agents.
 *
 * @property string $id
 * @property string $sender_did
 * @property string $receiver_did
 * @property A2ATaskState $state
 * @property string|null $skill_id
 * @property array<string, mixed>|null $input
 * @property array<string, mixed>|null $output
 * @property array<int, mixed>|null $artifacts
 * @property array<string, mixed>|null $metadata
 * @property string|null $error_message
 * @property \Illuminate\Support\Carbon|null $created_at
 * @property \Illuminate\Support\Carbon|null $updated_at
 */
class A2ATask extends Model
{
    use HasUuids;

    protected $table = 'a2a_tasks';

    protected $fillable = [
        'sender_did',
        'receiver_did',
        'state',
        'skill_id',
        'input',
        'output',
        'artifacts',
        'metadata',
        'error_message',
    ];

    /**
     * @return array<string, string>
     */
    protected function casts(): array
    {
        return [
            'state' => A2ATaskState::class,
            'input' => 'array',
            'output' => 'array',
            'artifacts' => 'array',
            'metadata' => 'array',
        ];
    }

    /**
     * @param Builder<self> $query
     * @return Builder<self>
     */
    public function scopeForSender(Builder $query, string $did): Builder
    {
        return $query->where('sender_did', $did);
    }

    /**
     * @param Builder<self> $query
     * @return Builder<self>
     */
    public function scopeForReceiver(Builder $query, string $did): Builder
    {
        return $query->where('receiver_did', $did);
    }

    /**
     * @param Builder<self> $query
     * @return Builder<self>
     */
    public function scopeActive(Builder $query): Builder
    {
        return $query->whereNotIn('state', [
            A2ATaskState::COMPLETED->value,
            A2ATaskState::CANCELED->value,
            A2ATaskState::FAILED->value,
        ]);
    }

    public function isTerminal(): bool
    {
        return $this->state->isTerminal();
    }

    public function transitionTo(A2ATaskState $newState): void
    {
        if (! $this->state->canTransitionTo($newState)) {
            throw new \RuntimeException(
                "Cannot transition A2A task from '{$this->state->value}' to '{$newState->value}'"
            );
        }

        $this->update(['state' => $newState]);
    }
}
```

```php
<?php

declare(strict_types=1);

namespace App\Domain\AgentProtocol\Services;

use App\Domain\AgentProtocol\Enums\A2ATaskState;
use App\Domain\AgentProtocol\Models\A2ATask;
use Illuminate\Support\Facades\Log;
use RuntimeException;

class A2ATaskService
{
    /**
     * @param array<string, mixed>|null $input
     * @param array<string, mixed>|null $metadata
     */
    public function createTask(
        string $senderDid,
        string $skillId,
        ?array $input = null,
        ?array $metadata = null,
    ): A2ATask {
        $task = A2ATask::create([
            'sender_did' => $senderDid,
            'receiver_did' => (string) config('app.url'),
            'state' => A2ATaskState::SUBMITTED,
            'skill_id' => $skillId,
            'input' => $input,
            'metadata' => $metadata,
        ]);

        Log::info('a2a: Task created', [
            'task_id' => $task->id,
            'sender' => $senderDid,
            'skill' => $skillId,
        ]);

        return $task;
    }

    public function getTask(string $taskId): ?A2ATask
    {
        return A2ATask::find($taskId);
    }

    public function cancelTask(string $taskId): A2ATask
    {
        $task = A2ATask::findOrFail($taskId);

        if ($task->isTerminal()) {
            throw new RuntimeException("Task {$taskId} is in terminal state '{$task->state->value}' and cannot be canceled.");
        }

        $task->transitionTo(A2ATaskState::CANCELED);

        Log::info('a2a: Task canceled', ['task_id' => $taskId]);

        return $task->refresh();
    }

    /**
     * @return \Illuminate\Database\Eloquent\Collection<int, A2ATask>
     */
    public function listTasksForSender(string $senderDid, ?string $state = null): \Illuminate\Database\Eloquent\Collection
    {
        $query = A2ATask::forSender($senderDid)->orderBy('created_at', 'desc');

        if ($state !== null) {
            $query->where('state', $state);
        }

        return $query->limit(100)->get();
    }
}
```

- [ ] **Step 4: Create A2ATaskController**

```php
<?php

declare(strict_types=1);

namespace App\Http\Controllers\Api\AgentProtocol;

use App\Domain\AgentProtocol\Services\A2ATaskService;
use App\Http\Controllers\Controller;
use Illuminate\Http\JsonResponse;
use Illuminate\Http\Request;
use RuntimeException;

class A2ATaskController extends Controller
{
    public function __construct(
        private readonly A2ATaskService $taskService,
    ) {
    }

    public function send(Request $request): JsonResponse
    {
        $validated = $request->validate([
            'sender_did' => 'required|string|max:255',
            'skill_id' => 'required|string|max:128',
            'input' => 'nullable|array',
            'metadata' => 'nullable|array',
        ]);

        $task = $this->taskService->createTask(
            senderDid: $validated['sender_did'],
            skillId: $validated['skill_id'],
            input: $validated['input'] ?? null,
            metadata: $validated['metadata'] ?? null,
        );

        return response()->json(['data' => $task], 201);
    }

    public function get(string $taskId): JsonResponse
    {
        $task = $this->taskService->getTask($taskId);

        if ($task === null) {
            return response()->json(['error' => 'Task not found'], 404);
        }

        return response()->json(['data' => $task]);
    }

    public function cancel(string $taskId): JsonResponse
    {
        try {
            $task = $this->taskService->cancelTask($taskId);

            return response()->json(['data' => $task]);
        } catch (RuntimeException $e) {
            return response()->json(['error' => $e->getMessage()], 422);
        }
    }

    public function list(Request $request): JsonResponse
    {
        $senderDid = $request->query('sender_did', '');
        $state = $request->query('state');

        if ($senderDid === '') {
            return response()->json(['error' => 'sender_did query parameter required'], 400);
        }

        $tasks = $this->taskService->listTasksForSender(
            (string) $senderDid,
            $state !== null ? (string) $state : null,
        );

        return response()->json(['data' => $tasks]);
    }
}
```

```yaml
name: Zelta SDK Release

on:
  push:
    tags:
      - 'sdk-v*'

permissions:
  contents: read

jobs:
  publish:
    name: Publish to Packagist
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - name: Setup PHP
        uses: shivammathur/setup-php@v2
        with:
          php-version: '8.4'
          tools: composer

      - name: Install dependencies
        working-directory: packages/zelta-sdk
        run: composer install --no-dev

      - name: Run tests
        working-directory: packages/zelta-sdk
        run: vendor/bin/pest

      - name: Notify Packagist
        run: |
          curl -X POST "https://packagist.org/api/update-package?username=${{ secrets.PACKAGIST_USERNAME }}&apiToken=${{ secrets.PACKAGIST_TOKEN }}" \
            -d '{"repository":{"url":"https://github.com/FinAegis/core-banking-prototype-laravel"}}'
        continue-on-error: true

      - name: Extract version
        id: version
        run: echo "version=${GITHUB_REF#refs/tags/sdk-v}" >> $GITHUB_OUTPUT

      - name: Summary
        run: |
          echo "## Zelta SDK v${{ steps.version.outputs.version }} Published" >> $GITHUB_STEP_SUMMARY
```
